% 定义后备字体家族，用于西文
\definefallbackfamily[opensc][rm]            % 名称、类型
    [latinmodernroman]                              % 默认字体/字体家族
    [range={basiclatin,latinsupplement},force=yes]  % 强制用于西文
\definefallbackfamily[opensc][ss]            % 名称、类型
    [latinmodernsans]                              % 默认字体/字体家族
    [range={basiclatin,latinsupplement},force=yes]  % 强制用于西文
\definefallbackfamily[opensc][tt]            % 名称、类型
    [latinmodernmonolight]                              % 默认字体/字体家族 latinmodernmonolight latinmodernmonoproplight
    [range={basiclatin,latinsupplement},force=yes]  % 强制用于西文

% 定义字体家族
\definefontfamily [opensc][rm]   % 名称、类型： rm/serif, ss/sansserif, tt/teletype, hw/handwritten, cg/calligraphic, mm/mathematics
    [notoserifcjksc]                           % 默认字体/字体家族
    [
        bf={notosanscjkscregular},                    % 粗体
        it={lxgwwenkairegular},                     % 斜体
        bi={notosanscjkscbold},                  % 粗斜体
    ]
\definefontfamily[opensc][ss]
    [zhuquefangsongtechnicalpreview]
    [
        bf={notosanscjkscregular},
        it={lxgwwenkairegular},
        bi={notosanscjkscbold}
    ]
\definefontfamily[opensc][tt]
    [lxgwwenkai]
    [
        bf={notosanscjkscregular},
        it={zhuquefangsongregular},
        bi={notosanscjkscbold}
    ]
